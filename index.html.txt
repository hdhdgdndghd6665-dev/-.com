    // БЕСПЛАТНАЯ РУЛЕТКА
    const REWARDS_ROULETTE = [50, 100, 200, 500, 1000, 2000, 5000, 10000];
    const COLORS = ['#FF6B6B', '#4ECDC4', '#45B7D1', '#96CEB4', '#FFEEAD', '#D4A5A5', '#9B59B6', '#F39C12'];
    let rouletteCanvas = null, ctx = null;
    let currentAngle = 0;
    let spinning = false;
    
    function drawWheel(angleDeg = 0) {
        if(!ctx || !rouletteCanvas) return;
        const size = rouletteCanvas.width;
        const center = size/2;
        const radius = size/2 - 5;
        const angleStep = (Math.PI * 2) / REWARDS_ROULETTE.length;
        let startAngle = angleDeg * Math.PI / 180;
        ctx.clearRect(0, 0, size, size);
        for(let i=0; i<REWARDS_ROULETTE.length; i++) {
            const endAngle = startAngle + angleStep;
            ctx.beginPath();
            ctx.fillStyle = COLORS[i % COLORS.length];
            ctx.moveTo(center, center);
            ctx.arc(center, center, radius, startAngle, endAngle);
            ctx.fill();
            ctx.save();
            ctx.translate(center, center);
            ctx.rotate(startAngle + angleStep/2);
            ctx.fillStyle = '#fff';
            ctx.font = 'bold 16px "Segoe UI"';
            ctx.fillText(REWARDS_ROULETTE[i], radius/1.8, 8);
            ctx.restore();
            startAngle = endAngle;
        }
        ctx.beginPath();
        ctx.arc(center, center, 20, 0, Math.PI*2);
        ctx.fillStyle = '#333';
        ctx.fill();
        ctx.strokeStyle = '#fff';
        ctx.lineWidth = 3;
        ctx.beginPath();
        ctx.moveTo(center, 10);
        ctx.lineTo(center-8, 25);
        ctx.lineTo(center+8, 25);
        ctx.fillStyle = '#FFD966';
        ctx.fill();
    }
    
    function spinWheel() {
        if(spinning) return;
        let lastTime = getLastFreeCaseTime(currentUser.username);
        if(Date.now() - lastTime < 5*3600000) { alert("Кейс ещё не восстановился! Подождите."); return; }
        spinning = true;
        const spinDuration = 2000;
        const startTime = performance.now();
        const startAngle = currentAngle;
        const targetSpins = 8 + Math.random() * 8;
        const finalAngleDelta = targetSpins * 360 + Math.random() * 360;
        function animate(now) {
            const elapsed = now - startTime;
            let t = Math.min(1, elapsed / spinDuration);
            const easeOut = 1 - Math.pow(1-t, 3);
            const angle = startAngle + finalAngleDelta * easeOut;
            currentAngle = angle % 360;
            drawWheel(currentAngle);
            if(t < 1) requestAnimationFrame(animate);
            else {
                spinning = false;
                const stopAngle = (360 - (currentAngle % 360)) % 360;
                const sectorSize = 360 / REWARDS_ROULETTE.length;
                let sectorIndex = Math.floor(stopAngle / sectorSize) % REWARDS_ROULETTE.length;
                const reward = REWARDS_ROULETTE[sectorIndex];
                addTokens(currentUser.username, reward);
                setLastFreeCaseTime(currentUser.username, Date.now());
                alert(`🎉 Рулетка остановилась на ${reward} токенов! Вы получили ${reward} токенов.`);
                updateTokenDisplay();
                renderFreeCase();
            }
        }
        requestAnimationFrame(animate);
    }

        let last = getLastFreeCaseTime(currentUser.username);

    const NFT_REWARDS = [100, 200, 500, 1000, "NFT"];
    const NFT_COLORS2 = ['#FF6B6B', '#4ECDC4', '#45B7D1', '#96CEB4', '#AA6DC9'];
    let nftCanvas = null, nftCtx = null;
    let nftAngle = 0;
    let nftSpinning = false;
    const nftImageUrl = "https://s00.yaplakal.com/pics/pics_original/5/6/9/20206965.jpg";
    
    function drawNFTWheel(angleDeg = 0) {
        if(!nftCtx || !nftCanvas) return;
        const size = nftCanvas.width;
        const center = size/2;
        const radius = size/2 - 5;
        const angleStep = (Math.PI * 2) / NFT_REWARDS.length;
        let startAngle = angleDeg * Math.PI / 180;
        nftCtx.clearRect(0, 0, size, size);
        for(let i=0; i<NFT_REWARDS.length; i++) {
            const endAngle = startAngle + angleStep;
            nftCtx.beginPath();
            nftCtx.fillStyle = NFT_COLORS2[i % NFT_COLORS2.length];
            nftCtx.moveTo(center, center);
            nftCtx.arc(center, center, radius, startAngle, endAngle);
            nftCtx.fill();
            nftCtx.save();
            nftCtx.translate(center, center);
            nftCtx.rotate(startAngle + angleStep/2);
            nftCtx.fillStyle = '#fff';
            nftCtx.font = 'bold 14px "Segoe UI"';
            let txt = NFT_REWARDS[i] === "NFT" ? "NFT" : NFT_REWARDS[i];
            nftCtx.fillText(txt, radius/1.8, 8);
            nftCtx.restore();
            startAngle = endAngle;
        }
        nftCtx.beginPath();
        nftCtx.arc(center, center, 20, 0, Math.PI*2);
        nftCtx.fillStyle = '#333';
        nftCtx.fill();
    }
    
    function spinNFTWheel() {
        if(nftSpinning) return;
        let lastSpin = getNFTLastSpin(currentUser.username);
        if(Date.now() - lastSpin < 3*3600000) {
            alert("NFT-рулетка доступна раз в 3 часа! Подождите.");
            return;
        }
        nftSpinning = true;
        const spinDuration = 2000;
        const startTime = performance.now();
        const startAngle = nftAngle;
        const targetSpins = 6 + Math.random() * 6;
        const finalAngleDelta = targetSpins * 360 + Math.random() * 360;
        function animate(now) {
            const elapsed = now - startTime;
            let t = Math.min(1, elapsed / spinDuration);
            const easeOut = 1 - Math.pow(1-t, 3);
            const angle = startAngle + finalAngleDelta * easeOut;
            nftAngle = angle % 360;
            drawNFTWheel(nftAngle);
            if(t < 1) requestAnimationFrame(animate);
            else {
                nftSpinning = false;
                const stopAngle = (360 - (nftAngle % 360)) % 360;
                const sectorSize = 360 / NFT_REWARDS.length;
                let sectorIndex = Math.floor(stopAngle / sectorSize) % NFT_REWARDS.length;
                const reward = NFT_REWARDS[sectorIndex];
                setNFTLastSpin(currentUser.username, Date.now());
                if(reward === "NFT") {
                    let modal = document.createElement('div');
                    modal.className = 'modal';
                    modal.innerHTML = `<div class="modal-card" style="text-align:center;"><h3>🎨 Вы выиграли NFT!</h3><img src="${nftImageUrl}" style="max-width:100%; border-radius:20px;"><br><button class="btn-small closeNftModal">Закрыть</button></div>`;
                    document.body.appendChild(modal);
                    modal.querySelector('.closeNftModal').onclick = () => modal.remove();
                } else {
                    addTokens(currentUser.username, reward);
                    alert(`🎉 Вы выиграли ${reward} токенов!`);
                }
                updateTokenDisplay();
                renderNFTroulette();
            }
        }
        requestAnimationFrame(animate);
    }

    function registerUser(login, displayName, password, emoji){
        let users=getUsers();
        if(users.find(x=>x.username===login)) throw new Error("Логин занят");
        if(users.find(x=>x.displayName===displayName)) throw new Error("Игровой ник занят");
        let newId = Math.floor(Math.random()*900000)+100000;
        while(users.find(u=>u.id===newId) || newId===7777777) newId = Math.floor(Math.random()*900000)+100000;
        let nu={id:newId, username:login, displayName:displayName, password:password, emoji:emoji, coins:100, power:1, defense:1, isVerified:false, isBanned:false, headerImage:'', voiceIntro:null, avatarImage:null, createdAt:Date.now()};
        users.push(nu); saveUsers(users);
        let fd=getFriendsData(); if(!fd[login]) fd[login]={friends:[],incoming:[],outgoing:[]}; saveFriendsData(fd);
        addTokens(login,0);
        return nu;
    }
    function loginUser(u,p){
        let user=getUsers().find(x=>x.username===u);
        if(!user) throw new Error("Пользователь не найден");
        if(user.isBanned) throw new Error("Вы забанены");
        if(user.password!==p) throw new Error("Неверный пароль");
        return user;
    }

    function initEmojiPicker() {
        const container = document.getElementById('emojiPickerReg');
        if(!container) return;
        container.innerHTML = '';
        allEmojis.forEach(em => {
            const span = document.createElement('span');
            span.innerText = em;
            span.className = 'emoji-item';
            span.onclick = () => {
                document.querySelectorAll('#emojiPickerReg .emoji-item').forEach(el => el.classList.remove('selected'));
                span.classList.add('selected');
